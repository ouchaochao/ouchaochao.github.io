MySQL Crash Course 1

Written by Chaochao Ou with ♥ on September 9, 2019 in SQL

以下是看MySQL必知必会这本书总结的一些知识点，备忘。

检索数据

# 检索单个列
SELECT prod_name FROM products;

# 检索多个列
SELECT prod_id, prod_name, prod_price FROM products;

# 检索所有列
SELECT * FROM products;

# 检索不同行（相同行只会显示一次）
SELECT DISTINCT prod_id FROM products;

# 限制结果
SELECT prod_name FROM products LIMIT 5; # 返回5行

SELECT prod_name FROM products LIMIT 5,5; # 返回第五行开始的5行

排序检索数据

# 排序数据
SELECT prod_name FROM products ORDER BY prod_name; # 升序（A-Z）

# 按多个列排序
SELECT prod_id, prod_name, prod_price FROM products ORDER BY prod_price, prod_name;

# 指定排序方向
SELECT prod_name FROM products ORDER BY prod_name DESC; # 降序（Z-A）

SELECT prod_id, prod_name, prod_price FROM products ORDER BY prod_price DESC, prod_name;

过滤数据

# 使用WHERE子句
SELECT prod_name FROM products WHERE prod_price = 2.5;
操作符 说明
= 等于
<> 不等于
!= 不等于
< 小于
<= 小于等于
> 大于
>= 大于等于
BETWEEN 在指定的两个值之间 
# 使用BETWEEN
SELECT prod_name FROM products WHERE prod_price BETWEEN 2 AND 5;

# 空值检查
SELECT prod_name FROM products WHERE prod_price IS NULL;

数据过滤

# AND
SELECT prod_name FROM products WHERE prod_price = 2.5 AND prod_id = 1001;

# OR
SELECT prod_name FROM products WHERE prod_price = 2.5 OR prod_price = 3.0;

# 注意计算次序
SELECT prod_name FROM products WHERE (prod_price = 2.5 OR prod_price = 3.0) AND prod_id = 1001;

# IN
SELECT prod_name FROM products WHERE prod_price IN (2.5,3.0) ORDER BY prod_name;

# NOT
SELECT prod_name FROM products WHERE prod_price NOT IN (2.5,3.0) ORDER BY prod_name;

用通配符进行过滤

# LIKE %
SELECT prod_name FROM products WHERE prod_name LIKE '%jet%'; # %表示任何字符出现任意次数

SELECT prod_name FROM products WHERE prod_name LIKE 'j%t';

# _ 匹配单个任意字符
SELECT prod_name FROM products WHERE prod_name LIKE 'j_t';

用正则表达式进行搜索

# REGEXP MySQL中的正则表达式匹配不区分大小写
SELECT prod_name FROM products WHERE prod_id REGEXP '1001' ORDER BY prod_name;

# BINARY 使得正则表达式区分大小写
SELECT prod_name FROM products WHERE prod_name REGEXP BINARY 'AbCd' ORDER BY prod_name;

# . 表示匹配任意一个字符
SELECT prod_name FROM products WHERE prod_name REGEXP '.001' ORDER BY prod_name;

# OR
SELECT prod_name FROM products WHERE prod_id REGEXP '1001 | 1002' ORDER BY prod_name;

# 匹配几个字符中之一: []
SELECT prod_name FROM products WHERE prod_id REGEXP '[123]001' ORDER BY prod_name;
# 等价于
SELECT prod_name FROM products WHERE prod_id REGEXP '1|2|3 001' ORDER BY prod_name;

# 匹配范围
SELECT prod_name FROM products WHERE prod_id REGEXP '[1-9]001' ORDER BY prod_name;

# 匹配特殊字符加 \\
SELECT prod_name FROM products WHERE prod_name REGEXP '\\.' ORDER BY prod_name;
元字符 说明
\\f 换页
\\n 换行
\\r 回车
\\t 制表
\\v 纵向制表 
# 匹配字符类（多个实例）
SELECT prod_name FROM products WHERE prod_name REGEXP '\\([0-9]sticks?\\)' ORDER BY prod_name; # ?的存在使得 stick 和 sticks 能同时存在，它匹配0或1个匹配
元字符 说明
* 0或多个匹配
+ 1或多个匹配
? 0或1个匹配
{n} 指定数目的匹配
{n,} 不少于指定数目的匹配
{n,m} 匹配数目范围[n,m]，m<255 
# 匹配字符类
SELECT prod_name FROM products WHERE prod_id REGEXP '[[:digits:]]{4}' ORDER BY prod_name;
# [[:digits:]]{4} 等价于 [0-9][0-9][0-9][0-9]
字符类 说明
[:alnum:] [a-zA-Z0-9]
[:alpha:] [a-zA-Z]
[:blank:] 空格和制表 [\\t]
[:cntrl:] ASCII 控制字符（ASCII 0到31和127）
[:digit:] [0-9]
[:graph:] 与[:print:]相同，但不包括空格
[:lower:] [a-z]
[:print:] 任意可打印字符
[:punct:] 不在[:alnum:]和[:cntrl:]中的任意字符
[:space:] 包括空格在内的空白字符[\\f\\n\\r\\t\\v]
[:upper:] [A-Z]
[:xdigit:] 任意16进制数字[a-fA-F0-9] 
# 定位符
SELECT prod_name FROM products WHERE prod_name REGEXP '^[0-9]' ORDER BY prod_name;
# ^ 表示文本的开始位置进行检索
元字符 说明
^ 文本的开始
$ 文本的结尾
[[:<:]] 词的开始
[[::>]] 词的结尾

创建计算字段

# 拼接
SELECT Concat(prod_name,'(', prod_price, ')') FROM products ORDER BY prod_name

# RTrim 去掉右边的空格， LTrim 去掉左边的空格， Trim去掉两边的空格
SELECT Concat(RTrim(prod_name),'(', RTrim(prod_price), ')') 
FROM products ORDER BY prod_name

# 使用别名 AS
SELECT Concat(RTrim(prod_name),'(', RTrim(prod_price), ')') 
AS prod_name_price
FROM products ORDER BY prod_name
